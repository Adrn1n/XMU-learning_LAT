		class NLTKNGramLM:
		    def __init__(self, n, seed):
		        self.n = n
		        self.rng = random.Random(seed)
		        self.vocab = set()
		        self.unigram = Counter()
		        self.context_counts = defaultdict(Counter)
		
		    @staticmethod
		    def __get_max_freq_chars(counter):
		        max_freq = max(counter.values())
		        return [ch for ch, freq in counter.items() if freq == max_freq]
		
		    def get_predict(self, history, _):
		        history = list(history)
		        context = (
		            tuple(history[-(self.n - 1) :])
		            if len(history) >= self.n - 1
		            else tuple(["<s>"] * (self.n - 1 - len(history)) + history)
		        )
		        if context in self.context_counts:
		            return self.rng.choice(
		                self.__get_max_freq_chars(self.context_counts[context])
		            )
		        elif self.unigram:
		            return self.rng.choice(self.__get_max_freq_chars(self.unigram))
		        else:
		            return "<unk>"
		
		    def train(self, lines):
		        for line in lines:
		            chars = list(line.strip())
		            self.vocab.update(chars)
		            self.unigram.update(chars)
		            padded = ["<s>"] * (self.n - 1) + chars
		            for ng in ngrams(padded, self.n):
		                context = tuple(ng[:-1])
		                next_char = ng[-1]
		                self.context_counts[context][next_char] += 1

		class CharDataset(Dataset):
		    def __init__(self):
		        super().__init__()
		        self.itos = ["<pad>", "<s>", "<unk>"]
		        self.stoi = {sp: i for i, sp in enumerate(self.itos)}
		        self.data = []
		
		    def __len__(self):
		        return len(self.data)
		
		    def __getitem__(self, idx):
		        return self.data[idx]
		
		    def add_data(self, lines, context_len):
		        for line in lines:
		            padded = ["<s>"] * context_len + list(line.strip())
		            for i in range(context_len, len(padded)):
		                x, y = padded[i - context_len : i], padded[i]
		                if y not in self.stoi:
		                    self.itos.append(y)
		                    self.stoi[y] = len(self.itos) - 1
		                self.data.append(
		                    (
		                        torch.tensor([self.stoi.get(c) for c in x], dtype=torch.long),
		                        torch.tensor(self.stoi[y], dtype=torch.long),
		                    )
		                )
		        return len(self.itos)
		
		    def itos_get(self, i):
		        return self.itos[i] if i < len(self.itos) else "<unk>"
		
		    def stoi_get(self, c):
		        return self.stoi.get(c, self.stoi["<unk>"])
		
		    def vocab_size(self):
		        return len(self.itos)
		
		
		class LSTMLM(nn.Module):
		    def __init__(self, device, emb_dim, hidden_dim, num_layers, seed):
		        super().__init__()
		        self.device = device
		        self.embdim = emb_dim
		        self.hiddendim = hidden_dim
		        self.lstm = nn.LSTM(
		            input_size=emb_dim,
		            hidden_size=hidden_dim,
		            num_layers=num_layers,
		            batch_first=True,
		        )
		        self.rng = random.Random(seed)
		        self.data = CharDataset()
		        self.emb = nn.Embedding(self.data.vocab_size(), emb_dim, padding_idx=0)
		        self.fc = nn.Linear(hidden_dim, self.data.vocab_size())
		        self.to(device)
		
		    def forward(self, x):
		        lstm_out, _ = self.lstm(self.emb(x))
		        return self.fc(lstm_out[:, -1, :])
		
		    def fit(self, lines, context_len, batch_size, optim, lr, epochs, criterion):
		        vocab_size = self.data.vocab_size()
		        if vocab_size < self.data.add_data(lines, context_len):
		            emb = nn.Embedding(self.data.vocab_size(), self.embdim, padding_idx=0).to(
		                self.device
		            )
		            with torch.no_grad():
		                if self.emb is not None:
		                    emb.weight[:vocab_size, :] = self.emb.weight
		            self.emb = emb
		            fc = nn.Linear(self.hiddendim, self.data.vocab_size()).to(self.device)
		            with torch.no_grad():
		                if self.fc is not None:
		                    fc.weight[:vocab_size, :] = self.fc.weight
		                    fc.bias[:vocab_size] = self.fc.bias
		            self.fc = fc
		        train_loader = DataLoader(self.data, batch_size, True)
		        optimizer = optim(self.parameters(), lr=lr)
		        self.train()
		        losses = []
		        for ep in range(epochs):
		            losses.append(0)
		            for x, y in tqdm(train_loader, desc=f"Epoch {ep+1}/{epochs}"):
		                optimizer.zero_grad()
		                loss = criterion(self(x.to(self.device)), y.to(self.device))
		                loss.backward()
		                optimizer.step()
		                losses[-1] += loss.item() * x.size(0)
		            losses[-1] /= len(self.data)
		            print(f"Epoch {ep+1}/{epochs}, Loss: {losses[-1]:.4f}")
		        _, ax = plt.subplots()
		        ax.plot(losses, marker="o", label="loss")
		        ax.set_xlabel("Epoch")
		        ax.set_ylabel("Loss")
		        ax.set_title("Training Loss")
		        ax.grid()
		        ax.legend()
		        plt.show()
		
		    @staticmethod
		    def __filling(history, context_len):
		        return (
		            history[-context_len:]
		            if len(history) >= context_len
		            else ["<s>"] * (context_len - len(history)) + history
		        )
		
		    def predict(self, history, context_len):
		        self.eval()
		        with torch.no_grad():
		            return self.forward(
		                torch.tensor(
		                    [
		                        [
		                            self.data.stoi_get(c)
		                            for c in self.__filling(history, context_len)
		                        ]
		                    ],
		                    dtype=torch.long,
		                    device=self.device,
		                )
		            )
		
		    def get_predict(self, history, context_len):
		        logits = self.predict(history, context_len).squeeze()
		        ids = (logits == torch.max(logits)).nonzero(as_tuple=True)[-1]
		        return self.data.itos_get(self.rng.choice(ids.cpu().tolist()))
